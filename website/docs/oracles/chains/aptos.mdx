```bash env title="aptos.env"
CHAIN="aptos"
# Aptos Config
APTOS_RPC_URL=""
APTOS_PID=""
# Oracle Config
APTOS_ORACLE_KEY=""
HEARTBEAT_INTERVAL=15
# Task Runner Config
TASK_RUNNER_SOLANA_RPC=""
GCP_CONFIG_BUCKET="" # Optional, for private queues
# Keypair Config, one-of required
FS_PAYER_SECRET_PATH=""
DOCKER_PAYER_SECRET=""
GOOGLE_PAYER_SECRET_PATH=""
AMAZON_PAYER_SECRET_PATH=""
# Monitoring Config
METRICS_EXPORTER="" # Optional
PAGERDUTY_EVENT_KEY="" # Optional
VERBOSE=1 # Optional
```

## Aptos Config

<table>
  <thead>
    <tr>
      <th>Env Variable</th>
      <th>Definition</th>
    </tr>
  </thead>
  <tbody>
    <tr>
      <td>CHAIN</td>
      <td>
        <b>
          <u>Required to be set to 'aptos'</u>
        </b>
        <br />
        <b>Type</b> - aptos / near / solana*
        <br />
        <b>Default</b> - solana
        <br />
        <b>Description</b> - Dictates which chain to listen for on-chain events
        <br />
      </td>
    </tr>
    <tr>
      <td>APTOS_RPC_URL</td>
      <td>
        <b>
          <u>Required</u>
        </b>
        <br />
        <b>Type</b> - URL
        <br />
        <b>Description</b> - Aptos RPC URL. Must not be rate limited
      </td>
    </tr>
    <tr>
      <td>APTOS_PID</td>
      <td>
        <b>
          <u>Required</u>
        </b>
        <br />
        <b>Type</b> - HexString
        <br />
        <b>Description</b> - On-chain address of the deployed Switchboard V2
        program
      </td>
    </tr>
  </tbody>
</table>

## Oracle Config

<table>
  <thead>
    <tr>
      <th>Env Variable</th>
      <th>Definition</th>
    </tr>
  </thead>
  <tbody>
    <tr>
      <td>APTOS_ORACLE_KEY</td>
      <td>
        <b>
          <u>Required</u>
        </b>
        <br />
        <b>Type</b> - Public Key
        <br />
        <b>Description</b> - Public key of the oracle account that has been
        granted permissions to use an oracle queue <br />
      </td>
    </tr>
    <tr>
      <td>HEARTBEAT_INTERVAL</td>
      <td>
        <b>
          <u>Optional</u>
        </b>
        <br />
        <b>Type</b> - Number (seconds)
        <br />
        <b>Default</b> - 30
        <br />
        <b>Description</b> - Seconds between oracle heartbeats. Queues have
        different oracle heartbeat requirements. Recommended value is 15
      </td>
    </tr>
  </tbody>
</table>
